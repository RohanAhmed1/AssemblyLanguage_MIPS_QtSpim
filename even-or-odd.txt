#check the integer is even or odd

.data
inp: .asciiz "Enter some positive intger: "
out1: .asciiz "The number is "
out2: .asciiz "even"
out3: .asciiz "odd"

.text

.globl main

.ent main

main:

    # print input message 
    li $v0, 4
    la $a0, inp
    syscall

    # get int input
    li $v0, 5
    syscall
    move $t0, $v0

    # print output 1

    li $v0, 4
    la $a0, out1
    syscall
    

    #calculations
    li $t1, 2
    div $t0, $t1
 
    mfhi $t2
 
    bne $t2, $0, odd
    
    li $v0, 4
    la $a0, out2
    syscall

    j end
 

    odd:
    li $v0, 4
    la $a0, out3
    syscall

    end:
    jr $ra
    
   




    

    



	
 
.end main
