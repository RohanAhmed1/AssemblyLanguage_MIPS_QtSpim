# arithmetic calcualtion of input number and fixed number


.data
inp1: .asciiz "Enter some integer value for x followed by enter: "
inp2: .asciiz "Enter some integer value for y followed by enter: "
inp3: .asciiz "Enter some integer value for z followed by enter: "
out: .asciiz "Result of x - y + z - 91 is: "

.text

.globl main

.ent main

main:

    # print input message 1
    li $v0, 4
    la $a0, inp1
    syscall

    # get int input
    li $v0, 5
    syscall
    move $t0, $v0


    # print input message 2
    li	$v0, 4
    la $a0, inp2
    syscall

    # get int input
    li $v0, 5
    syscall
    move $t1, $v0



    # print input message 3
    li	$v0, 4
    la $a0, inp3
    syscall

	
    # get int input and
    li $v0, 5
    syscall
    move $t2, $v0
    

    #calculations
    sub $t0, $t0, $t1
    add $t0, $t0, $t2
    addi $s0, $t0, -91

    



	
    # print output

    li $v0, 4
    la $a0, out
    syscall

    li $v0, 1
    move $a0, $s0
    syscall

  

    jr $ra

.end main
